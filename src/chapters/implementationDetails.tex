% Summary of the DQN-based approach for adaptive traffic signal control at Thapathali
This project focuses on developing an intelligent traffic control system using \gls{dqn} to optimize traffic flow at the Thapathali Intersection in Kathmandu. By applying \gls{dqn}, the system dynamically adjusts traffic light phases based on real-time data such as vehicle density, waiting times, and congestion levels. The primary objective is to minimize vehicle wait times and enhance traffic throughput by learning and implementing the most efficient signal phases.

The system utilizes \gls{sumo} for simulating traffic scenarios and \gls{traci} to enable real-time interaction between the \gls{dqn} model and the traffic simulation. This integration allows the \gls{dqn} model to control traffic signal phases, assess the impact of its actions, and continuously improve the traffic flow based on the simulation feedback.

\section{Vehicle Flow Generation}
\label{sec:vehicle_flow_generation}

\begin{algorithm}
\caption{Perfect Vehicle Allocation with Remainder Tracking}
\label{alg:perfect_allocation}
\begin{algorithmic}[1]
\State \textbf{Input:}
\State $\quad vehicle\_counts$: Dictionary $\{vc_1: count_1, vc_2: count_2, \dots\}$
\State $\quad proportions$: Dictionary $\{od_1: p_1, od_2: p_2, \dots\}$
\State $\quad od\_names$: List of OD names $[od_1, od_2, \dots, od_N]$
\State $\quad prev\_remainders$: Dictionary $\{od_1: r_1, od_2: r_2, \dots\}$ (Optional)

\State \textbf{Output:}
\State $\quad allocation$: Dictionary $\{od_1: \{vc_1: count, \dots\}, \dots\}$
\State $\quad new\_remainders$: Dictionary $\{od_1: r_1, od_2: r_2, \dots\}$

\Procedure{AllocateIntervalPerfect}{$vehicle\_counts$, $proportions$, $od\_names$, $prev\_remainders$}
    \If{$prev\_remainders$ is None}
        \State Initialize $prev\_remainders \gets \{od: 0 \text{ for all } od \in od\_names\}$
    \EndIf
    
    \State Initialize $allocation \gets \{od: \{\} \text{ for all } od \in od\_names\}$
    \State Initialize $new\_remainders \gets \{od: 0 \text{ for all } od \in od\_names\}$
    
    \ForAll{$(vehicle\_class, total\_count) \in vehicle\_counts$}
        \If{$total\_count = 0$}
            \State \textbf{continue}
        \EndIf
        
        \ForAll{$od \in od\_names$}
            \State $exact \gets total\_count \times proportions[od] + prev\_remainders[od]$
            \State $allocated \gets \lfloor exact \rfloor$
            \State $remainder \gets exact - allocated$
            
            \State $allocation[od][vehicle\_class] \gets allocated$
            \State $new\_remainders[od] \gets remainder$
        \EndFor
    \EndFor
    
    \State \Return $(allocation, new\_remainders)$
\EndProcedure
\end{algorithmic}
\end{algorithm}

\begin{algorithm}
\caption{Batch Perfect Allocation using piecewise linear interpolation}
\label{alg:batch_perfect}
\begin{algorithmic}[1]
\Procedure{AllocateVehiclesPerfect}{$vehicle\_counts$, $proportions$, $od\_names$}
    \State Initialize $result \gets \{od: \{vc: 0 \text{ for all } vc\} \text{ for all } od \in od\_names\}$
    
    \ForAll{$(vehicle\_class, total\_count) \in vehicle\_counts$}
        \If{$total\_count = 0$}
            \State \textbf{continue}
        \EndIf
        
        \State $targets \gets \{\}$ \Comment{Exact target allocations}
        \ForAll{$od \in od\_names$}
            \State $targets[od] \gets total\_count \times proportions[od]$
        \EndFor
        
        \State \Comment{Step 1: Allocate integer parts}
        \State $allocated \gets \{\}$
        \State $fractions \gets \{\}$
        \ForAll{$od \in od\_names$}
            \State $allocated[od] \gets \lfloor targets[od] \rfloor$
            \State $fractions[od] \gets targets[od] - allocated[od]$
        \EndFor
        
        \State $total\_allocated \gets \sum_{od \in od\_names} allocated[od]$
        \State $remaining \gets total\_count - total\_allocated$
        
        \If{$remaining > 0$}
            \State \Comment{Step 2: Sort ODs by fractional part (descending)}
            \State $sorted\_ods \gets \text{SORT}(fractions.items(), \text{by value descending})$
            
            \For{$i \gets 0$ \textbf{to} $\min(remaining, |od\_names|) - 1$}
                \State $od \gets sorted\_ods[i].key$
                \State $allocated[od] \gets allocated[od] + 1$
            \EndFor
        \EndIf
        
        \State \Comment{Store final allocation for this vehicle class}
        \ForAll{$od \in od\_names$}
            \State $result[od][vehicle\_class] \gets allocated[od]$
        \EndFor
    \EndFor
    
    \State \Return $result$
\EndProcedure
\end{algorithmic}
\end{algorithm}

\begin{algorithm}
\caption{Mathematical Formulation of largest remainder method for vehicle allocation}
\label{alg:mathematical_formulation}
\textbf{AllocateIntervalPerfect:}
\begin{align*}
\text{Given:} & \quad v_c \text{ (vehicles of class } c\text{), } p_n \text{ (proportion for OD } n\text{), } r_n^{prev} \text{ (previous remainder)} \\
\text{Compute:} & \quad exact_n = v_c \cdot p_n + r_n^{prev} \\
& \quad a_n = \lfloor exact_n \rfloor \\
& \quad r_n^{new} = exact_n - a_n \\
\text{Result:} & \quad \sum_{n} a_n = v_c \text{ (when summed over all classes with remainder tracking)}
\end{align*}

\textbf{AllocateVehiclesPerfect (Largest Remainder Method):}
\begin{align*}
\text{Given:} & \quad v_c \text{ (total vehicles of class } c\text{), } p_n \text{ (proportion for OD } n\text{)} \\
\text{Step 1:} & \quad a_n^{(1)} = \lfloor v_c \cdot p_n \rfloor \\
& \quad f_n = v_c \cdot p_n - a_n^{(1)} \\
\text{Step 2:} & \quad \text{Sort } \{f_n\} \text{ in descending order} \\
\text{Step 3:} & \quad a_n = a_n^{(1)} + 
\begin{cases}
1 & \text{if } n \text{ is among top } R \text{ ODs} \\
0 & \text{otherwise}
\end{cases} \\
\text{Where:} & \quad R = v_c - \sum_{n} a_n^{(1)}
\end{align*}

\textbf{Conservation Property:}
\[
\sum_{n=1}^{N} a_n = v_c \quad \forall c
\]
\end{algorithm}

\label{subsec:map_net_xml}

\label{subsec:routes_rou_xml}
This file defines the vehicle routes and traffic flow in the simulation. It specifies how many vehicles will be on the road and the routes they take through the network.

\label{subsec:vtypes_xml}
This file specifies the characteristics of different vehicle types used in the simulation. Each \texttt{<vType>} tag describes a vehicle's properties, such as length, width, acceleration, deceleration, maximum speed, and other factors relevant to vehicle behavior in the simulation.

    \item \textbf{accel:} The vehicle's acceleration in \unit{m/s^2}.
    \item \textbf{decel:} The vehicle's deceleration rate in \unit{m/s^2}.

    \item \textbf{minGap:} The minimum gap between vehicles in meters.
    \item \textbf{guiShape:} Specifies the visual representation of the vehicle.
    \item \textbf{sigma:} Influences the vehicle's randomness in lane-changing behavior.
    \item \textbf{tau:} Represents the reaction time of the vehicle.
\end{itemize}

\subsection{\texttt{flows.xml}}
\label{subsec:flows_xml}
In our simulation, the \texttt{departLane="best"} parameter was employed to ensure that vehicles are assigned to the most optimal lane when they enter the network. This helps to improve the overall efficiency of the traffic flow by minimizing congestion and ensuring that vehicles are distributed across available lanes in an optimal manner.

Additionally, a uniform vehicle generation pattern was applied, with vehicles entering the network in 15-minute intervals. This ensures that traffic flow is consistent and realistic, mirroring typical traffic patterns observed during peak hours while maintaining a balanced vehicle entry rate across the simulation period.

\subsection{\texttt{simulation.sumocfg} (SUMO Configuration File)}
\label{subsec:sumocfg}
The \gls{sumo} configuration file defines the simulation parameters, such as which files to use (network, route, vtype, flows, and add), the simulation duration, and various settings for the simulation itself.
The lateral-resolution parameter in the \gls{sumo} configuration file (\texttt{.sumocfg}) is used to define the resolution at which the lanes are subdivided during the simulation. It specifies the width of the lane in terms of the vehicle's lateral position. This value is particularly important for controlling the precision of lane-changing behavior, vehicle positioning, and simulation accuracy in \gls{sumo}.

To ensure high accuracy in vehicle positioning and lane-changing behavior, a lateral resolution of 0.2 meters was used in the \gls{sumo} simulation.

\section{Environment Wrapper Implementation}
\label{sec:env_wrapper}

The SumoEnvironment class provided the interface between the SUMO simulation and the reinforcement learning agent. The reset() method initialised each episode by restarting the SUMO simulation and resetting all internal state variables. The step(action) method executed the selected traffic signal action over a 15-second decision window, collecting rewards at each simulation step and returning the averaged reward, next state, and episode metrics upon completion. The \_get\_state() method extracted the 21-dimensional state vector comprising PCU-weighted vehicle loads on origin and outgoing lanes, maximum waiting times on stopline lanes, and a one-hot encoding of the current active phase. The \_calculate\_reward() method computed the reward by comparing stopline PCU against outgoing PCU discharge and adding a PCU-weighted accumulated waiting time penalty. The traffic light phases were controlled through SUMO's TraCI interface, with phase transitions managed internally by the environment.

The state representation used 21 dimensions: 7 origin lane PCU values reflecting approaching traffic volume, 6 stopline lane maximum waiting times capturing queue severity at the signal, 5 outgoing lane PCU values reflecting current discharge flow through the intersection, and 3 phase indicators encoded as one-hot vectors representing the currently active phase (P1, P2, P3). The action space consisted of 2 discrete actions - STAY and NEXT - operating over a cyclic sequence of three signal phases.

The reward function followed Max-Pressure theory. For each incoming lane, pressure was calculated as incoming vehicle count minus average outgoing vehicle count. The total reward was the negative sum of all lane pressures, encouraging congestion reduction.

\subsection{Traffic Signal Control Strategies}

In the fixed-time control approach, the signal operates on a constant cycle length of 60 seconds, where the phases P0, P1, and P2 are activated sequentially with predefined time allocations. This method does not consider real-time traffic conditions and follows a repetitive cycle regardless of traffic demand, making it simple to implement but less efficient under varying traffic conditions.

In contrast, the \gls{dqn}-based control approach introduces adaptive signal operation. In this method, the system evaluates traffic conditions at regular intervals of 15 seconds and decides whether to maintain the current phase or switch to another phase based on the observed traffic state. This allows the signal timing to adjust dynamically according to traffic demand, helping to reduce vehicle delays, queue lengths, and overall congestion at the intersection.

To simulate downstream backspill, a random-timer traffic signal with durations of 30, 45, or 60 seconds is placed at the end of each outgoing route to restrict vehicle movement. The signal enforces a 3-second yellow phase whenever the movement changes from green to red, which is deducted from the total green time. Meanwhile, the main (or Maternity) road operates on a fixed-timer signal with a 45-second green and 15-second red in a cyclic pattern in both fixed-timer and \gls{dqn}-based modes.

% Neural network architecture and learning mechanics of the DQN agent
\section{DQN Agent Implementation}
\label{sec:dqn_agent}

The neural network processed 21-dimensional state inputs through three hidden layers: 128 neurons in the first layer, 64 neurons in the second layer, and 32 neurons in the third layer, all using ReLU activation. The output layer produced 2 Q-values, one for each possible action - STAY and NEXT.

Two networks were maintained: a Q-network for action selection that updated continuously during training, and a target network for stable learning targets that updated every 10 episodes by copying the Q-network weights directly. Experience replay stored up to 10,000 transitions in a circular buffer and randomly sampled mini-batches of 64 transitions for each training step, breaking temporal correlations between consecutive experiences and improving learning stability.

$\epsilon$-greedy exploration started with fully random actions ($\epsilon = 1.0$) and gradually shifted to exploiting learned Q-values as $\epsilon$ decayed by a factor of 0.985 per episode, reaching a minimum of $\epsilon = 0.01$ by approximately episode 300. Q-learning updates computed current Q-values from the Q-network and target Q-values using the Bellman equation - rewards plus discounted maximum future Q-values from the target network with discount factor $\gamma = 0.95$. Mean squared error between the current and target Q-values was minimised during training using the Adam optimiser with a learning rate of 0.001.

% Episode-level training loop, checkpoint saving, and epsilon decay schedule
\section{Training Pipeline Implementation}
\label{sec:training_pipeline}

Training ran for 300 episodes. Each episode followed this sequence: reset simulation, observe initial state, select action using $\epsilon$-greedy policy, apply 3-second yellow phase when switching to next phase, execute 15-second green phase, collect reward averaged over the action duration, observe next state, store transition experience in replay buffer, train on random mini-batch of 64 if buffer contained sufficient samples, and update state for the next decision step.

Between episodes, $\epsilon$ decayed by multiplying by 0.985, gradually reducing random exploration in favour of learned policy exploitation. Every 10 episodes the target network updated its weights by copying directly from the Q-network, stabilising the learning targets. Model checkpoints were saved every 5 episodes to allow recovery and evaluation at intermediate training stages. Outputs included periodic checkpoints, a final trained model saved upon completion of all 300 episodes, and training visualisations showing reward, loss, average waiting time, throughput, and epsilon decay across episodes.

\section{Configuration Management}
\label{sec:config_management}

A central configuration file managed all parameters: training settings (episodes = 300, learning rate = 0.001, batch size = 64), DQN hyperparameters ($\epsilon$ decay = 0.985, discount factor $\gamma$ = 0.95, replay buffer size = 10,000, target network update frequency = 10 episodes), traffic light identifiers and phase definitions for P1, P2 and P3, lane definitions for origin, stopline and outgoing lanes, PCU values for each vehicle type present at Thapathali T1, reward weighting factor $\lambda$ = 0.0004, SUMO simulation file paths, and action durations (yellow transition: 3 seconds, green phase: 15 seconds per decision step). Centralising all parameters in a single configuration file ensured consistency across training, evaluation and comparison scripts and allowed hyperparameter adjustments without modifying the core environment or agent code.


\section{Metric Calculation}
\label{sec:metric_calculation}

Three primary metrics were tracked throughout training and evaluation to assess intersection performance.

Average Waiting Time was calculated by collecting the waiting time of all vehicles present on the incoming route lanes at regular intervals throughout each episode. At each metric collection step, the mean waiting time across all observed vehicles was accumulated and divided by the total number of collection steps to produce the episode-level average waiting time in seconds.

Average Queue Length was calculated by recording the number of halting vehicles across all incoming route lanes at each metric collection step. A vehicle was considered halting if it was either fully stopped below the SUMO default threshold of 0.1 m/s or slow-crawling below 1.5 m/s. The mean halting count across all monitored lanes was accumulated and divided by the total number of collection steps to produce the episode-level average queue length in vehicles.

Throughput per Wait Second was calculated as a composite efficiency metric by dividing the total episode throughput by the average waiting time, quantifying how many vehicles were processed per second of average delay experienced at the intersection:

$$\text{Throughput per Wait Second} = \frac{\text{Total Throughput (vehicles)}}{\text{Average Waiting Time (seconds)}}$$

A higher value indicates a more efficient intersection operation, with more vehicles cleared per unit of delay experienced, allowing direct comparison between the DQN agent and the fixed timer baseline across different traffic conditions.
